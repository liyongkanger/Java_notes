## ac847图中点的层次

### 题目描述

给定一个n个点m条边的有向图，图中可能存在重边和自环。

所有边的长度都是1，点的编号为1~n。

请你求出1号点到n号点的最短距离，如果从1号点无法走到n号点，输出-1。


#### 样例
**输入**
```
4 5
1 2
2 3
3 4
1 3
1 4
```
**输出**
```
1
```

----------

```c++
/*
1到n的最短路 边长都为 1 此时用bfs扩展到最后一个求得最短路
*/
#include<iostream>
#include<cstring>
#include<algorithm>
#include<queue>

using namespace std;
const int N = 100010;
int n, m;
int h[N], e[N], ne[N], idx;
int d[N];

void add(int a, int b){
    e[idx] = b;
    ne[idx] = h[a];
    h[a] = idx ++;
}
int bfs(){
    //初始化距离数组
    memset(d, -1, sizeof d);
    //定义一个队列
    queue<int> q;
    //1到1的距离为0
    d[1] = 0;
    q.push(1);
    
    while(q.size()){
        int t = q.front();
        q.pop();
        
        for(int i = h[t]; i != -1; i = ne[i]){
            int j = e[i];
            //如果0到j的距离没有更新 更新距离
            if(d[j] == -1){
                d[j] = d[t]  + 1;
                q.push(j);
            }
        }
    }
    return d[n];
}

int main(){
    
    scanf("%d%d", &n, &m);
    
    memset(h, -1, sizeof h);
    
    for(int i = 0; i < m; i ++){
        int a, b;
        scanf("%d%d", &a, &b);
        add(a, b);
    }
    
    cout << bfs() << endl;
    
    return 0;
}
```
